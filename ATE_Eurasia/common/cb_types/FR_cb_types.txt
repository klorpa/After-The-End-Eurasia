fr_chaotic_revolt = {
	name = CB_NAME_CHAOTIC_REVOLT
	war_name = WAR_NAME_CHAOTIC_REVOLT
	sprite = 7
	truce_days = 3650
	hostile_against_others = yes
	can_ask_to_join_war = no
	apply_short_occ_mod = no # Do not apply the 'recently_conquered' modifier to Holdings
	apply_long_occ_mod = no # Do not apply the 'new administration' modifier to Holdings
	can_call_allies = no
	allow_whitepeace = no

	can_use = {
		FROM = {
			NOT = { culture = monster }
		}
		ROOT = {
			NOT = { culture = monster }
		}
		ROOT = {
			has_character_flag = chaotic_revolter
		}
	}

	on_add = {
		hidden_tooltip = { fire_haruspicy_event_effect = yes }
		if = {
			limit = { defender = { is_offmap_governor = shou_lung } }
			attacker = {
				sound_effect = china_angered_emperor
				detract_grace_super_huge_effect = yes
			}
		}
	}

	on_success = {
		any_player = {
			limit = {
				OR = {
					dynasty = FROM
					any_liege = { character = FROM }
					is_within_diplo_range = FROM
				}
			}
			narrative_event = { id = fr_uprising.13 }
		}
		if = {
			limit = {
				ROOT = {
					trait = patron_hoar
					is_rival = FROM
				}
			}
			ROOT = { piety = 40 }
		}
		FROM = {
			if = {
				limit = { NOT = { has_dlc = "Zeus" } }
				any_demesne_title = { # All titles
					if = {
						limit = {
							OR = {
								tier = king
								tier = emperor
							}
							NOT = { has_law = crown_authority_0 }
						}
						add_law = crown_authority_0
					}
				}
			}
		}

		ROOT = {
			prestige = 200
			piety = 100
			hidden_tooltip = {
				if = {
					limit = { is_landed = false }
					occupy_minors_of_occupied_settlements = FROM
					gain_all_occupied_titles = { who = FROM type = revolt }
				}
			}
			participation_scaled_prestige = 200
			participation_scaled_piety = 100
			hidden_tooltip = {
				if = {
					limit = {
						has_nickname = no
					}
					random_list = {
						10 = { give_nickname = nick_the_realmbreaker }
						10 = { give_nickname = nick_crownsplitter }
					}
				}
			}
		}
		any_attacker = {
			limit = { NOT = { character = ROOT } }
			hidden_tooltip = {
				participation_scaled_prestige = 200
				participation_scaled_piety = 100
			}
		}
		any_attacker = {
			any_vassal = {      #Vassals of vassals go first to reduce scoping
				hidden_effect = {
					any_demesne_title = {
						limit = {
							higher_tier_than = COUNT
							NOR = {
								controls_religion = yes
								clan = yes
							}
						}
						unsafe_destroy_landed_title = THIS
					}
				}
			}
            hidden_tooltip = {      # Every county a ruler
                any_vassal = {
                    any_demesne_title = {
                        limit = { tier = COUNT is_primary_holder_title = no }
                        location = {
                            create_character = {
                                religion = THIS
                                culture = THIS
                                random_traits = yes
                            }
                            new_character = {
                                save_event_target_as = popular_revolt_new_character
                                PREVPREV = {
                                    usurp_title_plus_barony_if_unlanded = { target = event_target:popular_revolt_new_character type = revolt }
                                }
                                set_defacto_liege = event_target:popular_revolt_new_character
                                any_demesne_title = {
                                    limit = { tier = BARON }
                                    refill_holding_levy = yes
                                }
                            }
                        }
                    }
                }
            }
			hidden_tooltip = {
				hidden_effect = {
					any_demesne_title = {
						limit = {
							higher_tier_than = DUKE
							NOR = {
								controls_religion = yes
								clan = yes
							}
						}
						unsafe_destroy_landed_title = THIS
					}
				}
			}
			hidden_tooltip = {        # Crack all titles except one Duke title
				hidden_effect = {
					any_demesne_title = {
						limit = {
							higher_tier_than = COUNT
							is_primary_holder_title = no
							NOR = {
								controls_religion = yes
								clan = yes
							}
						}
						unsafe_destroy_landed_title = THIS
					}
				}
			}
			hidden_tooltip = {
				any_demesne_title = {
					limit = { tier = COUNT location = { is_capital = no } }
					location = {
						create_character = {
							religion = THIS
							culture = THIS
							random_traits = yes
						}
						new_character = {
							save_event_target_as = popular_revolt_new_character
							PREVPREV = {
								usurp_title_plus_barony_if_unlanded = { target = event_target:popular_revolt_new_character type = revolt }
							}
							set_defacto_liege = event_target:popular_revolt_new_character
							any_demesne_title = {
								limit = { tier = BARON }
								refill_holding_levy = yes
							}
						}
					}
				}
			}
			any_vassal = {          # Everyone goes free, except the defeated ruler
				limit = { higher_tier_than = BARON NOT = { has_character_flag = hogging_khan } }
				set_defacto_liege = this
			}
		}

		FROM = {
			prestige = -200
			any_vassal = {
				hidden_effect = {
					any_demesne_title = {
						limit = {
							higher_tier_than = COUNT
							NOR = {
								controls_religion = yes
								clan = yes
							}
						}
						unsafe_destroy_landed_title = THIS
					}
				}
			}
            hidden_tooltip = {
                any_vassal = {
                    any_demesne_title = {
                        limit = { tier = COUNT is_primary_holder_title = no }
                        location = {
                            create_character = {
                                religion = THIS
                                culture = THIS
                                random_traits = yes
                            }
                            new_character = {
                                save_event_target_as = popular_revolt_new_character
                                PREVPREV = {
                                    usurp_title_plus_barony_if_unlanded = { target = event_target:popular_revolt_new_character type = revolt }
                                }
                                any_demesne_title = {
                                    limit = { tier = BARON }
                                    refill_holding_levy = yes
                                }
                            }
                        }
                    }
                }
            }
			custom_tooltip = {      # Cracks all Duke titles of FROM as well to make them eligible for vassalisation by ROOT
				text = destabilization_shou_lung_success_tt
				hidden_effect = {
					any_demesne_title = {
						limit = {
							higher_tier_than = COUNT
							NOR = {
								controls_religion = yes
								clan = yes
							}
						}
						unsafe_destroy_landed_title = THIS
					}
				}
			}
			hidden_tooltip = {
				any_demesne_title = {
					limit = { tier = COUNT is_primary_holder_title = no }
					location = {
						create_character = {
							religion = THIS
							culture = THIS
							random_traits = yes
						}
						new_character = {
							save_event_target_as = popular_revolt_new_character
							PREVPREV = {
								usurp_title_plus_barony_if_unlanded = { target = event_target:popular_revolt_new_character type = revolt }
							}
							any_demesne_title = {
								limit = { tier = BARON }
								refill_holding_levy = yes
							}
						}
					}
				}
            }

			set_defacto_liege = ROOT
			imprison = ROOT

			hidden_tooltip = {
				if = {
					limit = {
						has_nickname = no
					}
					random_list = {
						10 = { give_nickname = nick_the_unready }
						10 = { give_nickname = nick_the_ill_ruler }
					}
				}
			}
		}

		hidden_tooltip = {
			ROOT = {
				disband_event_forces = popular_revolters
                add_rival = FROM
			}
			FROM = {
				character_event = { id = fr_uprising.23 }
			}
			any_attacker = {
				character_event = { id = fr_uprising.24 }
			}
		}
	}

	on_fail = {
		any_player = {
			limit = {
				OR = {
					dynasty = FROM
					any_liege = { character = FROM }
					is_within_diplo_range = FROM
				}
			}
			narrative_event = { id = fr_uprising.14 }
		}
		ROOT = {
			prestige = -200
            any_realm_character = {
				if = {
					limit = {
						trait = peasant_leader
					}
                    hidden_tooltip = {
                        clear_wealth = yes
                        imprison = FROM
                        add_character_modifier = {
                            name = broken_spirit
                            days = -1
                        }
                    }
				}
            }
			hidden_tooltip = {
				any_demesne_title = {
					if = {
						limit = { temporary = yes tier = EMPEROR }
						unsafe_destroy_landed_title = THIS
					}
				}
			}
		}
		FROM = {
			prestige = 100
			participation_scaled_prestige = 100
			opinion = {
                modifier = opinion_traitor
                who = ROOT
			}
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { participation_scaled_prestige = 100 }
		}
		any_attacker = {
			hidden_tooltip = {
				any_vassal = {
					if = {
						limit = { has_character_flag = designated_rebel_nomad_ally }
						set_defacto_liege = FROM
					}
				}
			}
		}
		any_attacker = {
			hidden_tooltip = { set_defacto_liege = FROM }
		}
		hidden_tooltip = {
			ROOT = {
				disband_event_forces = popular_revolters
				add_rival = FROM
			}
			FROM = {
				character_event = { id = fr_uprising.23 }
			}
			any_attacker = {
				character_event = { id = fr_uprising.24 }
			}
		}
	}

	on_reverse_demand = {
		any_player = {
			limit = {
				OR = {
					dynasty = FROM
					any_liege = { character = FROM }
					is_within_diplo_range = FROM
				}
			}
			narrative_event = { id = fr_uprising.14 }
		}
		if = {
			limit = {
				FROM = {
					trait = patron_hoar
					is_rival = ROOT
				}
			}
			FROM = { piety = 40 }
		}
		ROOT = {
			hidden_tooltip = {
				any_demesne_title = {
					if = {
						limit = { temporary = yes tier = EMPEROR }
						unsafe_destroy_landed_title = THIS
					}
				}
			}
		}
		any_attacker = {
			hidden_tooltip = {
				any_vassal = {
					if = {
						limit = { has_character_flag = designated_rebel_nomad_ally }
						set_defacto_liege = FROM
					}
				}
			}
			hidden_tooltip = { set_defacto_liege = FROM }
		}
		ROOT = {
			prestige = -200
            any_realm_character = {
				if = {
					limit = {
						trait = peasant_leader
					}
                    hidden_tooltip = {
                        clear_wealth = yes
                        imprison = FROM
                        add_character_modifier = {
                            name = broken_spirit
                            days = -1
                        }
                    }
				}
            }
		}
		FROM = {
			prestige = 200
			opinion = {
                modifier = opinion_traitor
                who = ROOT
			}
		}
		ROOT = {
			imprison = FROM
			disband_event_forces = popular_revolters
			add_rival = FROM
		}
		hidden_tooltip = {
			FROM = {
				character_event = { id = fr_uprising.23 }
			}
			any_attacker = {
				character_event = { id = fr_uprising.24 }
			}
		}
	}

	on_reverse_demand_title = {
		ROOT = { show_scope_change = no remove_claim = PREV }
	}

	attacker_ai_victory_worth = {
		factor = 100
	}

	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}

	defender_ai_defeat_worth = {
		factor = 150
	}
}
