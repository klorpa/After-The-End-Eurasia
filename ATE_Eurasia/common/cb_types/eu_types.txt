imperial_ottoman_restoration = {
	name = CB_NAME_OTTOMAN_RESTORATION
	war_name = WAR_NAME_OTTOMAN_RESTORATION
	sprite = 17
	truce_days = 3650
	is_permanent = yes
	check_de_jure_tier = DUKE # this scans all dejure duchies for the counties which are held by or vassals(or below) of selected character. Only valid if is_permanent = yes

	allowed_to_target_tributaries = no
	
	sort_priority = 1900

	can_use_gui = {
		check_if_crusader_trigger = yes
	}

	can_use = {
		ROOT = {
			has_landed_title = e_ottoman_empire
			independent = yes
			OR = { 
				religion = nizami
				religion = ruhi
			}
			NOT = { same_realm = FROM }
			mercenary = no
		}
		FROM = {
			NOT = { religion_group = muslim }
		}
	}

	can_use_title = { #more titles will be added when those places start existing
		OR = {
			#Sham
			de_jure_liege_or_above = k_jordan
			de_jure_liege_or_above = k_lebanon
			de_jure_liege_or_above = k_ard_al_muqaddis
			de_jure_liege_or_above = k_iraq
			de_jure_liege_or_above = k_syria
			#Arabia
			de_jure_liege_or_above = k_hijaz
			de_jure_liege_or_above = k_bahrayn
			de_jure_liege_or_above = k_kuwait
			de_jure_liege_or_above = k_yemen
			#Misr/Egypt
			de_jure_liege_or_above = k_egypt
			#Maghrib
			de_jure_liege_or_above = k_libya
			de_jure_liege_or_above = k_tunisia
			de_jure_liege_or_above = k_algeria
			#Rumelia (Balkans)
			de_jure_liege_or_above = k_albania
			de_jure_liege_or_above = k_bulgaria
			de_jure_liege_or_above = k_libya
			#Yugoslavia
			de_jure_liege_or_above = k_croatia
			de_jure_liege_or_above = k_bosnia
			de_jure_liege_or_above = k_slovenia
			de_jure_liege_or_above = k_serbia
			de_jure_liege_or_above = k_montenegro
			#Austria-Hungary
			de_jure_liege_or_above = k_hungary
			de_jure_liege_or_above = k_austria
			#Crimea
			de_jure_liege_or_above = k_crimea
		}
	}
	
	is_valid = {
		ROOT = {
			independent = yes
			OR = { 
				religion = nizami
				religion = ruhi
			}
			NOT = { same_realm = FROM }
		}
	}

	on_add = {
		hidden_tooltip = { fire_haruspicy_event_effect = yes }
		if = {
			limit = { defender = { is_offmap_governor = offmap_china } }
			attacker = {  
				sound_effect = china_angered_emperor
				detract_grace_super_huge_effect = yes 
			}
		}
	}
	
	on_success = {
		ROOT = {
			participation_scaled_prestige = 100
		}
		any_attacker = {
			limit = { NOT = { character = ROOT } }
			hidden_tooltip = { 
				participation_scaled_prestige = 100
			}
		}
		
		FROM = { prestige = -100 }
	}

	on_success_title = {
		custom_tooltip = { 
			text = other_invasion_succ_tip
			hidden_tooltip = {
				ROOT = {
					vassalize_or_take_under_title = {
						title = PREV
						enemy = FROM
						type = invasion
					}
				}
			}
		}
		if = {
			limit = {
				holder_scope = { character = FROM }
			}
			usurp_title_only = { target = ROOT type = invasion }
		}

		hidden_tooltip = {
			check_if_reconquista_finished_effect = yes
			FROM = {
				landless_pope_effect = yes
			}
		}
	}

	on_fail = {
		ROOT = { prestige = -150 }
	}

	on_reverse_demand = {
		ROOT = {
			transfer_scaled_wealth = {
				to = FROM
				value = 4.0
			}
			ROOT = { prestige = -200 }
		}
		FROM = {
			participation_scaled_piety = 100
			participation_scaled_prestige = 200
			if = {
				limit = {
					uses_decadence = yes
				}
				participation_scaled_decadence = -10
			}
		}
		any_defender = {
			limit = { NOT = { character = FROM } }
			hidden_tooltip = { 
				participation_scaled_piety = 100
				participation_scaled_prestige = 200
				if = {
					limit = {
						uses_decadence = yes
					}
					participation_scaled_decadence = -10
				}
			}
		}
	}

	attacker_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 100
	}

	defender_ai_victory_worth = {
		factor = -1 # always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
}
