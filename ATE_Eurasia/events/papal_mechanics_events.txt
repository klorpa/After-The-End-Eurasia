namespace = papal

character_event = { #defensor fidei becomes weak
	id = papal.1
	religion = catholic
	desc = EVTDESC_papal_1
	picture = "GFX_evt_bishop"
	
	trigger = {
		religion_authority >= 0.65
		is_ruler = yes
		trait = defensor_fidei
		NOT = { has_alternate_start_parameter = { key = religion_names value = random } }
	}
	
	mean_time_to_happen = {
		years = 5
	}
	
	option = { #fuck all my power is gone
		name = EVTOPTA_papal_1
		prestige = -150
		add_character_modifier = { name = weak_defensor_fidei duration = -1 }
		remove_character_modifier = strong_defensor_fidei
	}
}
