# Jews Expelled
character_event = {
	id = SoA.105
	desc = EVTDESC_SoA_105
	picture = GFX_evt_jewish_market
	border = GFX_event_normal_frame_religion
	
	is_triggered_only = yes
	
	hide_from = yes
	
	immediate = {
		any_realm_lord = {
			limit = {
				religion_group = jewish_group
				ai = yes
			}
			any_demesne_title = {
				grant_title_no_opinion = ROOT
			}
		}
	}

	option = {
		name = EVTOPTA_SoA_105
		hidden_tooltip = {
			any_realm_lord = {
				limit = { 
					OR = {
						ai = no
						has_character_modifier = borrowed_from_jews
					}
				}
				character_event = { id = SoA.107 }
			}
		}
	}
}

character_event = {
	id = SoA.500
	desc = EVTDESC_SoA_500
	picture = GFX_evt_synagogue
	border = GFX_event_normal_frame_religion
	
	is_triggered_only = yes
	
	hide_from = yes
	
	option = {
		name = EVTOPTA_SoA_500
		any_vassal = {
			limit = {
				religion_group = jewish_group
				prisoner = no
				war = no
				NOT = { trait = incapable }
			}
			character_event = { id = SoA.501 days = 2 tooltip = EVTTOOLTIP_SoA_501 }
		}
		custom_tooltip = { text = passover_invitations_sent }
		if = {
			limit ={
				OR = { 
					religion = ivri
					culture = sefardi
				}
			}
			hidden_tooltip = { character_event = { id = SoA.509 days = 9 } }
		}
		hidden_tooltip = { character_event = { id = SoA.503 days = 10 } }
		hidden_tooltip = {
			random_list = {
				50 = { character_event = { id = SoA.506 days = 8  } } 
				50 = { 
					modifier = {
						factor = 0
						num_of_vassal <= 2
					}
					character_event = { id = SoA.507 days = 11 } 
				} 
				50 = { character_event = { id = SoA.508 days = 11 } } 
			}
		}
		hidden_tooltip = { character_event = { id = SoA.504 days = 12 } }
	}
