narrative_event = {
	id = anoa.5	
	title = EVTTITLE_swiss_invitation
	desc = EVTDESC_the_swiss_delegation
	picture = GFX_evt_bishop
	is_triggered_only = yes
	
	trigger = {
		has_landed_title = e_germany
		OR = {
			has_global_flag = swiss_desire_to_join
			has_global_flag = swiss_no_desire_to_join
		}
	}
	
	option = {
		name = accept_swiss_delegation
		trigger = {
			has_global_flag = swiss_desire_to_join
		}
		clr_global_flag = swiss_desire_to_join
		k_switzerland = {
			de_jure_liege = e_germany
			set_defacto_liege = e_germany
		}
		prestige = 1000
	}
	option = {
		name = declined_swiss_delegation
		trigger = {
			has_global_flag = swiss_no_desire_to_join
		}
		clr_global_flag = swiss_no_desire_to_join
		prestige = -1000
	}
}
