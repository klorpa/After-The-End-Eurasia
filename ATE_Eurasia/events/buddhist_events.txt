			factor = 50

			}
			modifier = {
				factor = 0.5
				age = 50
			}
			modifier = {
				factor = 0.7
				trait = craven
			}
			modifier = {
				factor = 0.5
				trait = gluttonous 
			}
			modifier = {
				factor = 0.5
				trait = cynical
			}
		}
	}
	option = {#power through

			limit = {
				NOT = {
					has_dlc = "Reapers"
				}
			}
			add_trait = ill
		}

			add_symptom_effect = yes
		}

		custom_tooltip  = {
			text = fast_sicknesss

				chance = 10

				}

				}

				}
