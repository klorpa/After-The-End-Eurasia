namespace = aventura


#Aventura iniciada
character_event = {
	id = aventura.1
	desc = EVTDESC_aventura1
	picture = GFX_evt_ship_voyage_hf
	border = GFX_event_normal_frame_religion
	is_triggered_only = yes
	
	immediate = {
		set_character_flag = on_aventura
	}
	
	#pay extra for the ship
	option = {
		name = EVTOPTA_aventura1
		wealth = -200
		set_character_flag = best_ship
		hidden_tooltip = {
			character_event = { id = aventura.2 days = 3}
		}
	}
	
	#normal price for ship
	option = {
		name = EVTOPTB_aventura1
		wealth = -100
		set_character_flag = good_ship
		hidden_tooltip = {
			character_event = { id = aventura.2 days = 3}
		}
	}

	#high stewardship pays low for the ship
	option = {
		name = EVTOPTC_aventura1
		tooltip_info = stewardship
		trigger = {
			stewardship = 12
		}
		wealth = -100
		set_character_flag = best_ship
		hidden_tooltip = {
			character_event = { id = aventura.2 days = 3}
		}
	}
	
	#actually I'd rather stay
	option = {
		name = EVTOPTD_aventura1
		clr_character_flag = on_aventura
	}

}
