namespace = titlegiving


#On Start Up Ya boys will get the correct title
character_event = {
	id = titlegiving.001
	hide_window = yes
	is_triggered_only = yes
	
		trigger = {
		government = theocracy_government
		religion = miaphysite 
		higher_tier_than = BARON	
		NOT = { has_landed_title = d_miaphysite_alexandria }
		}
	
	immediate = {
			set_special_character_title = PING_PRESBYTEROS
		
	}
}
