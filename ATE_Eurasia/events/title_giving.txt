namespace = titlegiving


#On Start Up Ya boys will get the correct title
character_event = {
	id = titlegiving.001
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		government = theocracy_government
			holder_scope = {
				higher_tier_than = BARON
				OR = {
				religion = miaphysite
				religion = sunni
				title = d_georgian_patriarch
					AND = {
						religion = mariavite
						NOT = { title = d_mariavite }
					}
				}				
			}
	}
	
	immediate = {
		if = {
			limit = { 
			   religion = miaphysite
			}
			set_special_character_title = PING_PRESBYTEROS
		}
		else_if = {
			limit = { 
				religion = mariavite
				holder_scope = {
				NOT = { title = d_mariavite }
				}
			}
			set_special_character_title = PING_PRIMAS
		}
		else_if = {
			limit = { 
			   religion = sunni
			}
			set_special_character_title = PING_GRAND_MUFTI
		}
		else_if = {
			limit = { 
				religion = orthodox
				holder_scope = {
				title = d_georgian_patriarch
				}
				{ title = d_georgian_patriarch status = yes }
			}
			set_special_character_title = PING_CATHOLICOS_PATRIARCH
		}
		else = {
			#Do nothing lmao
		}
	}
}
