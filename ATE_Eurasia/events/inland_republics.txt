namespace = inland
character_event = {#adds a titular county
	id = inland.1
	picture = GFX_evt_powerful_ruler
	desc = EVTDESC_become_oligarch
	
	only_rulers = yes
	only_playable = yes
	
	
	trigger = {
		government = playable_republic_government
		NOT = {
			any_title_under = {
				has_title_flag = inland_patrician
			}
		}
	}
	
	mean_time_to_happen = {
		months = 12
		modifier = {
			factor = 2
			primary_title = {
				any_direct_de_jure_vassal_title = {
					count = 1
					has_title_flag = inland_patrician
				}
			}
		}
		modifier = {
			factor = 2
			primary_title = {
				any_direct_de_jure_vassal_title = {
					count = 2
					has_title_flag = inland_patrician
				}
			}
		}
		modifier = {
			factor = 2
			primary_title = {
				any_direct_de_jure_vassal_title = {
					count = 3
					has_title_flag = inland_patrician
				}
			}
		}
		modifier = {
			factor = 2
			primary_title = {
				any_direct_de_jure_vassal_title = {
					count = 4
					has_title_flag = inland_patrician
				}
			}
		}
		modifier = {
			factor = 2
			primary_title = {
				any_direct_de_jure_vassal_title = {
					count = 5
					has_title_flag = inland_patrician
				}
			}
		}
		modifier = {
			factor = 2
			primary_title = {
				any_direct_de_jure_vassal_title = {
					count = 6
					has_title_flag = inland_patrician
				}
			}
		}
		modifier = {
			factor = 2
			primary_title = {
				any_direct_de_jure_vassal_title = {
					count = 7
					has_title_flag = inland_patrician
				}
			}
		}
	}
	option = {
		name = EVTOPTA_become_oligarch
		primary_title = {
			capital_scope = {
				ROOT = {
					create_title = {
						tier = COUNT
						name = "c_inland_republic"
						holder = THIS
						base_title = PREVPREV
						foa = oligarch
						landless = yes
					}
					new_title = {
						set_title_flag = inland_patrician
						set_name = "House of [Root.GetLastWordInDynastyName]"
						set_coa = PREVPREVPREV#primary title
						de_jure_liege = PREV
					}
				}
			}
		}
	}
}

character_event = {#removes the titular county
	id = inland.2
	is_triggered_only = yes
	
	picture = GFX_evt_powerful_ruler
	desc = EVTDESC_lose_oligrach
	
	only_rulers = yes
	only_playable = yes
	
	option = {
		name = EVTOPTA_lose_oligarch
		any_title = {
			limit = {
				has_title_flag = inland_republic
			}
			activate_title = {
				title = THIS
				status = no
			}
		}
	}
}
