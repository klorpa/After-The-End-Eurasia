	id = shia.5
	title = EVTNAME_al_waqil_1
	desc = EVTDESC_al_waqil_1
	picture = GFX_an_najaf
	
	only_independent = yes
	
	trigger = {
		NOT = { trait = waqil }
	}
	
	is_triggered_only = yes

		name = EVTOPTA_al_waqil_1
		add_trait = waqil
	}
}
